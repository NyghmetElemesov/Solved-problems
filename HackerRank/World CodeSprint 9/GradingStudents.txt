public class GradingStudents {
	static int grading(int grade) {
		// return the failing grade below 40
		if (grade < 38)
			return grade;
		else if (grade == 38 || grade == 39 || grade == 40)
			return 40;
		// return the pass grades
		if(grade%5>=3){
			if(grade%5==3)
				return grade+2;
			else if(grade%5==4)
				return grade+1;
			else if(grade%5==5)
				return grade+0;
		}
			
		
		return grade;
	}

	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		int n = in.nextInt();
		for (int a0 = 0; a0 < n; a0++) {
			int grade = in.nextInt();
			// your code goes here
			System.out.println(grading(grade));
		}
	}
}
